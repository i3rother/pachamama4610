003package aufgabe1;
004
005import java.util.Scanner;
006
007/**
008 * IntVarExtended zeigt den Umgang mit Variablen vom Typ int.
009 * &Uuml;bungsaufgabe 1 zur Programmiertechnik 1.
010 * @author Raphael Bregger
011 * @version 09.04.2019
012 */
013public final class IntVarExtended {
014    private IntVarExtended() { }
015
016    private static final Scanner EINGABE = new Scanner(System.in);
017
018    /**
019     * main ist der Startpunkt des Programms.
020     * @param args wird nicht verwendet.
021     */
022    public static void main(String[] args) {
023        final int min = 0x8000_0000;
024        final int max = 0x7fff_ffff;
025
026        //Variables
027        long firstInt;
028        long secondInt;
029        long operationResult = 0;
030        String operator;
031
032        //First message
033        System.out.printf("%s %d %s %d %s%n%s%n",
034            "Bitte geben sie zwei ganze Zahlen zwischen",
035            min, "und", max, "ein.", "Bitte geben Sie die erste Zahl ein:");
036
037        //Fetching first number
038        EINGABE.useDelimiter("\n");
039        while (!EINGABE.hasNextInt()) {
040            System.out.println("Das war keine zulässige Eingabe.");
041            EINGABE.next();
042        }
043        firstInt = EINGABE.nextInt();
044
045        //Fetching second number
046        System.out.println("Bitte geben sie die zweite Zahl ein:");
047        while (!EINGABE.hasNextInt()) {
048            System.out.println("Das war keine Zulässige Eingabe.");
049            EINGABE.next();
050        }
051        secondInt = EINGABE.nextInt();
052
053        //Fetching the operator
054        System.out.println("Bitte geben sie eine arithmetischen Operator ein:");
055        while (!(EINGABE.hasNext("\\+") || EINGABE.hasNext("\\-")
056            || EINGABE.hasNext("\\*") || EINGABE.hasNext("\\/")
057            || EINGABE.hasNext("%"))) {
058            System.out.println("Bitte einen gültigen Operator eingeben.");
059            EINGABE.next();
060        }
061        operator = EINGABE.next();
062
063        //Perform arithmetic operation
064        switch (operator) {
065        case "+": operationResult = firstInt + secondInt;
066                  break;
067        case "-": operationResult = firstInt - secondInt;
068                  break;
069        case "*": operationResult = firstInt * secondInt;
070                  break;
071        case "/": operationResult = firstInt / secondInt;
072                  break;
073        case "%": operationResult = firstInt % secondInt;
074                  break;
075        default:
076            System.out.println("Something went wrong");
077            break;
078        }
079
080        //Validate for integer dadatyp
081        if (operationResult <= max && operationResult >= min) {
082            System.out.printf("Das Ergebnis ist: %d%n", operationResult);
083        } else if (operationResult >= max) {
084            System.out.println("Das Ergebnis ist für den Datentyp "
085                + "\"Integer\" zu groß.");
086        } else {
087            System.out.println("Das Ergebnis ist für den Datentyp"
088                + " \"Integer\" zu klein.");
089        }
090    }
091}
